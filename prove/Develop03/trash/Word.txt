using System;
public class Prompt

    private static Reference _library = new Reference();
    private static Scripture _scripture = new Scripture();
    private static Random _random = new Random();
    private static string _joinedScripture = _library.GetScripture(_random.Next(0, _library.GetCount()-1));
    private static string[] seperatedScripture = _joinedScripture.Split("|");
    
    public void SetScripture()
    {
    _scripture.SetReference(seperatedScripture[0]);
    _scripture.SetVerse(seperatedScripture[1]);
    foreach (string verse in seperatedScripture[2].Split("$"))
    {
        _scripture.SetVerseText().Append<string>(verse);
    }
    }
    
    public void DisplayScripture()
    {  
    SetScripture();  
    Console.WriteLine($"{_scripture.GetBook()} {_scripture.GetChapter()} {_scripture.GetVerse()} {_scripture.GetVerseText()}");
    }
